#include<stdio.h>
#include<stdlib.h>
int main()
{
	int i=1,a,b;
	while(i!=0)
	{	printf("please input a year:");
		scanf("%d",&a);
		if(a%100==0)
			(a%400==0)?b=1:b=0;
		else
			(a%4==0)?b=1:b=0;
		if(b==1)
			printf("%d is a leap year!  i=%d\n",a,i);
		else

		i++;
	}
